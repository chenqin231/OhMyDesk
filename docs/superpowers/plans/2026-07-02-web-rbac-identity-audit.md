# WEB RBAC Identity Audit Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build dynamic WEB user management with fixed 4-role RBAC and make every remote session/audit record attributable to the real logged-in person.

**Architecture:** The server owns identity, roles, and authorization. The frontend consumes `/api/me` permissions for menu and route filtering, but every sensitive HTTP/WS operation is enforced again on the server. Existing random `admin-xxxx` websocket IDs remain transport route IDs only; audit identity comes from JWT-bound `AuthUser`.

**Tech Stack:** Rust/Axum/SQLx/SQLite/bcrypt/jsonwebtoken/ts-rs, React/Vite/Zustand/TypeScript, existing Docker deployment.

```rust
async fn add_column_if_missing(pool: &Db, table: &str, column: &str, ddl: &str) -> Result<(), sqlx::Error> {
    let rows = sqlx::query(&format!("PRAGMA table_info({table})"))
        .fetch_all(pool)
        .await?;
    let exists = rows.iter().any(|row| {
        use sqlx::Row;
        row.try_get::<String, _>("name").map(|name| name == column).unwrap_or(false)
    });
    if !exists {
        sqlx::query(ddl).execute(pool).await?;
    }
    Ok(())
}
```

In `connect()` after `raw_sql(ddl)` succeeds, call:

```rust
impl std::str::FromStr for Role {
    type Err = String;

    fn from_str(s: &str) -> Result<Self, Self::Err> {
        match s {
            "superadmin" => Ok(Role::Superadmin),
            "admin" => Ok(Role::Admin),
            "operator" => Ok(Role::Operator),
            "auditor" => Ok(Role::Auditor),
            _ => Err(format!("未知角色: {s}")),
        }
    }
}

impl UserStore {
    pub fn new(db: Db) -> Self {
        UserStore { db }
    }

    pub async fn bootstrap(&self, legacy: Option<(String, String)>) {
        let count: i64 = sqlx::query_scalar("SELECT COUNT(*) FROM users")
            .fetch_one(&self.db)
            .await
            .unwrap_or(0);
        if count > 0 {
            return;
        }
        let now = now_sec();
        let super_hash = crate::auth::hash_password("infogo123");
        let _ = self
            .create_with_hash("superadmin", &super_hash, Role::Superadmin, true, now)
            .await;
        if let Some((legacy_user, legacy_hash)) = legacy {
            let username = if legacy_user.trim().is_empty() || legacy_user == "superadmin" {
                "admin".to_string()
            } else {
                legacy_user.trim().to_string()
            };
            let _ = self
                .create_with_hash(&username, &legacy_hash, Role::Admin, true, now)
                .await;
        }
    }

    async fn create_with_hash(
        &self,
        username: &str,
        password_hash: &str,
        role: Role,
        enabled: bool,
        now: i64,
    ) -> Result<UserRecord, String> {
        let id = uuid::Uuid::new_v4().to_string();
        let username = username.trim();
        if username.is_empty() {
            return Err("用户名不能为空".into());
        }
        sqlx::query(
            "INSERT INTO users (id, username, password_hash, role, enabled, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?)",
        )
        .bind(&id)
        .bind(username)
        .bind(password_hash)
        .bind(role.as_str())
        .bind(if enabled { 1 } else { 0 })
        .bind(now)
        .bind(now)
        .execute(&self.db)
        .await
        .map_err(|e| format!("创建用户失败: {e}"))?;
        self.get_by_id(&id).await.ok_or_else(|| "创建用户后读取失败".into())
    }

    pub async fn create(
        &self,
        username: &str,
        password: &str,
        role: Role,
        enabled: bool,
    ) -> Result<UserRecord, String> {
        if password.is_empty() {
            return Err("密码不能为空".into());
        }
        let hash = crate::auth::hash_password(password);
        self.create_with_hash(username, &hash, role, enabled, now_sec()).await
    }

    pub async fn list(&self) -> Vec<UserRecord> {
        sqlx::query_as::<_, UserRow>(
            "SELECT id, username, password_hash, role, enabled, created_at, updated_at FROM users ORDER BY created_at ASC",
        )
        .fetch_all(&self.db)
        .await
        .unwrap_or_default()
        .into_iter()
        .filter_map(UserRecord::try_from_row)
        .collect()
    }

    pub async fn get_by_username(&self, username: &str) -> Option<UserRecord> {
        sqlx::query_as::<_, UserRow>(
            "SELECT id, username, password_hash, role, enabled, created_at, updated_at FROM users WHERE username = ?",
        )
        .bind(username)
        .fetch_optional(&self.db)
        .await
        .ok()
        .flatten()
        .and_then(UserRecord::try_from_row)
    }

    pub async fn get_by_id(&self, id: &str) -> Option<UserRecord> {
        sqlx::query_as::<_, UserRow>(
            "SELECT id, username, password_hash, role, enabled, created_at, updated_at FROM users WHERE id = ?",
        )
        .bind(id)
        .fetch_optional(&self.db)
        .await
        .ok()
        .flatten()
        .and_then(UserRecord::try_from_row)
    }
}

#[derive(sqlx::FromRow)]
struct UserRow {
    id: String,
    username: String,
    password_hash: String,
    role: String,
    enabled: i64,
    created_at: i64,
    updated_at: i64,
}

impl UserRecord {
    fn try_from_row(row: UserRow) -> Option<Self> {
        Some(UserRecord {
            id: row.id,
            username: row.username,
            password_hash: row.password_hash,
            role: row.role.parse().ok()?,
            enabled: row.enabled != 0,
            created_at: row.created_at,
            updated_at: row.updated_at,
        })
    }
}

fn now_sec() -> i64 {
    std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap()
        .as_secs() as i64
}
```

```rust
use crate::users::{Role, UserRecord, UserStore};
use std::sync::Arc;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Claims {
    pub sub: String,
    pub username: String,
    pub role: String,
    pub exp: i64,
}

pub struct Auth {
    users: Arc<UserStore>,
    secret: Vec<u8>,
}

impl Auth {
    pub fn new(secret: Vec<u8>, users: Arc<UserStore>) -> Self {
        Auth { users, secret }
    }

    pub async fn verify_login(&self, user: &str, pass: &str) -> Option<UserRecord> {
        let record = self.users.get_by_username(user).await?;
        if !record.enabled {
            return None;
        }
        if bcrypt::verify(pass, &record.password_hash).unwrap_or(false) {
            Some(record)
        } else {
            None
        }
    }

    pub fn issue_token(&self, user_id: &str, username: &str, role: Role, now: i64) -> String {
        let claims = Claims {
            sub: user_id.to_string(),
            username: username.to_string(),
            role: role.as_str().to_string(),
            exp: now + TOKEN_TTL_SECS,
        };
        encode(&Header::default(), &claims, &EncodingKey::from_secret(&self.secret)).unwrap_or_default()
    }

    pub async fn validate(&self, token: &str) -> Option<crate::http::AuthUser> {
        let claims = self.validate_token_only(token)?;
        let role: Role = claims.role.parse().ok()?;
        let user = self.users.get_by_id(&claims.sub).await?;
        if !user.enabled || user.username != claims.username || user.role != role {
            return None;
        }
        Some(crate::http::AuthUser {
            id: user.id,
            username: user.username,
            role: user.role,
        })
    }

    pub fn validate_token_only(&self, token: &str) -> Option<Claims> {
        let validation = Validation::new(Algorithm::HS256);
        decode::<Claims>(token, &DecodingKey::from_secret(&self.secret), &validation)
            .ok()
            .map(|d| d.claims)
    }
}
```

```rust
    State(s): State<HttpState>,
    user: AuthUser,
    axum::extract::Path(id): axum::extract::Path<String>,

) -> impl IntoResponse {

        return e.into_response();
    }
    let mut current = match s.users.get_by_id(&id).await {
        Some(u) => u,
        None => return (StatusCode::NOT_FOUND, Json(json!({ "error": "用户不存在" }))).into_response(),

            Ok(role) => role,
            Err(e) => return (StatusCode::BAD_REQUEST, Json(json!({ "error": e }))).into_response(),
        };
```

```tsx
    } catch (err) {
      setError(err instanceof Error ? err.message : "重置密码失败");
    }
  }

  return (
    <AppShell title="用户管理" online={online} total={endpoints.length}>
      <div className="grid gap-4">
        <Card>
          <CardHeader>
            <CardTitle>新增用户</CardTitle>
          </CardHeader>
          <CardContent>
            <form className="grid gap-3 md:grid-cols-[1fr_1fr_180px_auto]" onSubmit={handleCreate}>
              <Input value={username} onChange={(e) => setUsername(e.target.value)} placeholder="用户名" required />
              <Input value={password} onChange={(e) => setPassword(e.target.value)} placeholder="初始密码" type="password" required />
              <Select value={role} onValueChange={(v) => setRole(v as Role)}>
                <SelectTrigger><SelectValue /></SelectTrigger>
                <SelectContent>
                  {roles.map((r) => <SelectItem key={r.value} value={r.value}>{r.label}</SelectItem>)}
                </SelectContent>
              </Select>
              <Button type="submit">新增</Button>
            </form>
            {error && <p className="mt-3 text-sm text-destructive">{error}</p>}
          </CardContent>
        </Card>
        <Card>
          <CardHeader>
            <CardTitle>账号列表</CardTitle>
          </CardHeader>
          <CardContent className="overflow-x-auto">
            <table className="w-full text-sm">
              <thead>
                <tr className="border-b text-left text-muted-foreground">
                  <th className="py-2">用户名</th>
                  <th>角色</th>
                  <th>状态</th>
                  <th className="text-right">操作</th>
                </tr>
              </thead>
              <tbody>
                {users.map((u) => {
                  const locked = u.role === "superadmin";
                  return (
                    <tr key={u.id} className="border-b">
                      <td className="py-2 font-medium">{u.username}</td>
                      <td>
                        {locked ? "超级管理员" : (
                          <Select value={u.role} onValueChange={(v) => void handleRole(u, v as Role)}>
                            <SelectTrigger className="w-32"><SelectValue /></SelectTrigger>
                            <SelectContent>{roles.map((r) => <SelectItem key={r.value} value={r.value}>{r.label}</SelectItem>)}</SelectContent>
                          </Select>
                        )}
                      </td>
                      <td>{u.enabled ? "启用" : "停用"}</td>
                      <td className="space-x-2 text-right">
                        <Button variant="outline" size="sm" onClick={() => void handleReset(u)}>重置密码</Button>
                        <Button variant="outline" size="sm" disabled={locked} onClick={() => void handleToggle(u)}>
                          {u.enabled ? "停用" : "启用"}
                        </Button>
                      </td>
                    </tr>
                  );
                })}
              </tbody>
            </table>
          </CardContent>
        </Card>
      </div>
    </AppShell>
  );
}
```
